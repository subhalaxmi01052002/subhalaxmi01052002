<body>
    <button class="color-button" onclick="changeColor()">Click Me</button>

    <script>
        let isOrange = true;

        function changeColor() {
            const button = document.querySelector('.color-button');
            if (isOrange) {
                button.style.backgroundColor = '#7e7ee1';
            } else {
                button.style.backgroundColor = '#f87b33';
            }
            isOrange = !isOrange;
        }
    </script>
